You can enable in SDK 51 manually, ensure you set clear the bundler cache.

```json
{
  "experiments": {
    "reactCompiler": true
  }
}
```

... pass in custom settings ...

```js

module.exports = function (api) {
  api.cache(true);

  return {
    presets: [['babel-preset-expo', { 'babel-plugin-react-compiler': {
        ...
    } }]],
  };
};
```
